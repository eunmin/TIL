# 배열을 변환해서 배열 출력하기


```swift
override func viewDidLoad() {
        super.viewDidLoad()
        print(trans(["Enter uid123 Muzi", "Enter uid1244 Ryan", "Leave uid123 Muzi"]))
    }


    func trans(_ input : [String]) -> [String] {
        var final = [String]()
        for i in input {
            var a = i.components(separatedBy: " ")
            var result : String
            if a[0] == "Enter" {
                result = "입장했습니다"
            } else {
                result = "나갔습니다"
            }

            final.append("\(a[2])가(이) \(result)")

        }
        return final
    }
```
